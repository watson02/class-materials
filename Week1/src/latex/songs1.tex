\begin{itemize}
\item[]
\textbf{Algorithm:} Find Most Played Song(S)
\item[]
\textbf{Input:} A set of play counts associated with a variety of songs inside a playlist.
\item[]
\textbf{Output:} The most played song.
\item[]
\scalebox{1}{
\begin{algorithm}[H]
\begin{algorithmic}[1]
\STATE $temp \gets S[0]$
\STATE $res \gets 1$
\FOR{$i=1$ to $|S|$} 
\IF{$S[i] > temp$}
\STATE $temp \gets S[i]$
\STATE $res \gets i+1$
\ENDIF
\ENDFOR
\STATE \textbf{return} res;

\end{algorithmic}
\label{alg:seq}
\end{algorithm}
}
